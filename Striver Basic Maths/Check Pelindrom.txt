import java.util.Scanner;

public class Main {
	
	public static void main(String[] args) {
		// Write your code here
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		int temp = n;
		int rev=0; 

		while(n>0){
			int rem = n%10;
			rev = rev *10 + rem;
			n/=10;
		}
		System.out.println(rev==temp);
	}
}
