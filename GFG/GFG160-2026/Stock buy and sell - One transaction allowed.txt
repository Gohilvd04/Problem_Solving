class Solution {
  public:
    int maximumProfit(vector<int> &prices) {
        // code here
        int minPrice = INT_MAX;
        int n = prices.size();
        int maxProfit = 0;
        
        for(int i =0; i<n; i++){
            if(prices[i] < minPrice){
                minPrice = prices[i];
            }else{
                maxProfit = max(maxProfit, prices[i] - minPrice);
            }
        }
        return maxProfit;
    }
};
