class Solution {
  public:
    void pushZerosToEnd(vector<int>& arr) {
        // code here
        int n = arr.size();
        int index = 0;
        
        // Move all non-zero elements to the left
        for(int i =0; i<n; i++){
            if(arr[i] != 0){
                arr[index++] = arr[i];
            }
        }
        // add zeroes to remaining position
        while(index < n){
            arr[index++] = 0;
        }
    }
};
